# JS — 垃圾回收机制

# 可达性

值通过某种方式可访问，则这个值具有可达性，它们被保证存在于内存中

# 一个引用

// 将一个name为john的对象赋值给user全局变量
let user = {
  name: "John"
};

# 复杂引用

function marry(man, woman) {
  man.wife = woman;
  woman.husband = man;

  return {
    father: man,
    mother: woman
  }
}
